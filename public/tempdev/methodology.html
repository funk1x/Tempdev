import { queue } from "@infra/jobs";
const worker = async () => {
const task = await queue.next();
if (!task) return;
// normalize payload
const clean = sanitize(task.data);
const result = await process(clean);
notify(result.id);
};
export const config = {
retries: 3, timeout: 8000
};
queue.on("done", log);
setInterval(worker, 1200);
export default worker;
